# Solutions to Homework Problems

1. **Check if a string is a palindrome**
```cpp
#include <iostream>
#include <string>
bool isPalindrome(std::string s) {
    int left = 0, right = s.length() - 1;
    while (left < right) {
        if (s[left++] != s[right--]) return false;
    }
    return true;
}
int main() {
    std::string s = "racecar";
    std::cout << (isPalindrome(s) ? "Yes" : "No") << std::endl; // Output: Yes
    return 0;
}
```

2. **Reverse a string in-place**
```cpp
#include <iostream>
#include <string>
void reverseString(std::string& s) {
    int left = 0, right = s.length() - 1;
    while (left < right) {
        std::swap(s[left++], s[right--]);
    }
}
int main() {
    std::string s = "hello";
    reverseString(s);
    std::cout << s << std::endl; // Output: olleh
    return 0;
}
```

3. **Check if two strings are anagrams**
```cpp
#include <iostream>
#include <string>
#include <algorithm>
bool isAnagram(std::string s1, std::string s2) {
    if (s1.length() != s2.length()) return false;
    std::sort(s1.begin(), s1.end());
    std::sort(s2.begin(), s2.end());
    return s1 == s2;
}
int main() {
    std::string s1 = "listen", s2 = "silent";
    std::cout << (isAnagram(s1, s2) ? "Yes" : "No") << std::endl; // Output: Yes
    return 0;
}
```

4. **Find the first occurrence of a substring**
```cpp
#include <iostream>
#include <string>
int strStr(std::string haystack, std::string needle) {
    if (needle.empty()) return 0;
    for (int i = 0; i <= haystack.length() - needle.length(); i++) {
        if (haystack.substr(i, needle.length()) == needle) return i;
    }
    return -1;
}
int main() {
    std::string haystack = "hello", needle = "ll";
    std::cout << strStr(haystack, needle) << std::endl; // Output: 2
    return 0;
}
```

5. **Count occurrences of a character**
```cpp
#include <iostream>
#include <string>
int countChar(std::string s, char c) {
    int count = 0;
    for (char ch : s) {
        if (ch == c) count++;
    }
    return count;
}
int main() {
    std::string s = "banana";
    std::cout << countChar(s, 'a') << std::endl; // Output: 3
    return 0;
}
```

6. **Convert string to lowercase**
```cpp
#include <iostream>
#include <string>
std::string toLowerCase(std::string s) {
    for (char& c : s) {
        if (c >= 'A' && c <= 'Z') c += 32;
    }
    return s;
}
int main() {
    std::string s = "Hello";
    std::cout << toLowerCase(s) << std::endl; // Output: hello
    return 0;
}
```

7. **Check if string contains only digits**
```cpp
#include <iostream>
#include <string>
bool isDigitString(std::string s) {
    for (char c : s) {
        if (!isdigit(c)) return false;
    }
    return true;
}
int main() {
    std::string s = "12345";
    std::cout << (isDigitString(s) ? "Yes" : "No") << std::endl; // Output: Yes
    return 0;
}
```

8. **Remove all occurrences of a character**
```cpp
#include <iostream>
#include <string>
std::string removeChar(std::string s, char c) {
    int write = 0;
    for (int read = 0; read < s.length(); read++) {
        if (s[read] != c) s[write++] = s[read];
    }
    return s.substr(0, write);
}
int main() {
    std::string s = "hello";
    std::cout << removeChar(s, 'l') << std::endl; // Output: heo
    return 0;
}
```

9. **Longest common prefix**
```cpp
#include <iostream>
#include <vector>
#include <string>
std::string longestCommonPrefix(std::vector<std::string>& strs) {
    if (strs.empty()) return "";
    std::string prefix = strs[0];
    for (int i = 1; i < strs.size(); i++) {
        while (strs[i].find(prefix) != 0) {
            prefix = prefix.substr(0, prefix.length() - 1);
            if (prefix.empty()) return "";
        }
    }
    return prefix;
}
int main() {
    std::vector<std::string> strs = {"flower", "flow", "flight"};
    std::cout << longestCommonPrefix(strs) << std::endl; // Output: fl
    return 0;
}
```

10. **Valid palindrome ignoring non-alphanumeric**
```cpp
#include <iostream>
#include <string>
bool isValidPalindrome(std::string s) {
    int left = 0, right = s.length() - 1;
    while (left < right) {
        while (left < right && !isalnum(s[left])) left++;
        while (left < right && !isalnum(s[right])) right--;
        if (tolower(s[left++]) != tolower(s[right--])) return false;
    }
    return true;
}
int main() {
    std::string s = "A man, a plan, a canal: Panama";
    std::cout << (isValidPalindrome(s) ? "Yes" : "No") << std::endl; // Output: Yes
    return 0;
}
```
