#include <iostream>  // For input/output operations
#include <string>    // For string data type
#include <iomanip>   // For formatting output
using namespace std;  // Avoids repeated std:: for readability

// Define Student class to encapsulate student data
class Student {
public:
    int id;                  // Student ID (unique)
    string name;             // Student name
    float grade;             // Average grade (0.0 to 100.0)

    // Constructor to initialize student data
    Student(int studentId = 0, string studentName = "", float studentGrade = 0.0)
        : id(studentId), name(studentName), grade(studentGrade) {}

    // Function to display student details
    void display() const {
        cout << left << setw(10) << id 
             << setw(20) << name 
             << setw(10) << fixed << setprecision(2) << grade 
             << endl;
    }
};

// Define StudentDatabase class to manage array of students
class StudentDatabase {
private:
    static const int MAX_STUDENTS = 100;  // Maximum number of students
    Student students[MAX_STUDENTS];       // Array to store student objects
    int count;                            // Current number of students

    // Helper function to check if ID already exists
    bool isIdUnique(int id) const {
        for (int i = 0; i < count; i++) {  // Loop through all students
            if (students[i].id == id) {
                return false;              // ID found, not unique
            }
        }
        return true;                       // ID is unique
    }

public:
    // Constructor to initialize count
    StudentDatabase() : count(0) {}

    // Function to add a student
    void addStudent(int id, const string& name, float grade) {
        // Validate inputs
        if (count >= MAX_STUDENTS) {
            cout << "Error: Database full!" << endl;
            return;
        }
        if (!isIdUnique(id)) {
            cout << "Error: ID " << id << " already exists!" << endl;
            return;
        }
        if (grade < 0.0 || grade > 100.0) {
            cout << "Error: Grade must be between 0.0 and 100.0!" << endl;
            return;
        }

        // Add student to array using reference to avoid copying
        students[count] = Student(id, name, grade);
        count++;  // Increment student count
        cout << "Student added successfully!" << endl;
        students[count-1].display();  // Show added student's details
    }

    // Function to display all students
    void displayAll() const {
        if (count == 0) {
            cout << "No students in database!" << endl;
            return;
        }
        // Print header
        cout << left << setw(10) << "ID" 
             << setw(20) << "Name" 
             << setw(10) << "Grade" << endl;
        cout << string(40, '-') << endl;
        // Loop through students and display each
        for (int i = 0; i < count; i++) {
            students[i].display();
        }
    }

    // Function to search for a student by ID
    void searchById(int id) const {
        for (int i = 0; i < count; i++) {  // Loop through students
            if (students[i].id == id) {    // Check if ID matches
                cout << "Student found:" << endl;
                cout << left << setw(10) << "ID" 
                     << setw(20) << "Name" 
                     << setw(10) << "Grade" << endl;
                cout << string(40, '-') << endl;
                students[i].display();
                return;
            }
        }
        cout << "Student with ID " << id << " not found!" << endl;
    }
};

// Function to display menu and get user choice
int getMenuChoice() {
    int choice;
    cout << "\nStudent Database Management System\n";
    cout << "1. Add Student\n";
    cout << "2. Display All Students\n";
    cout << "3. Search by ID\n";
    cout << "4. Exit\n";
    cout << "Enter choice (1-4): ";
    cin >> choice;
    return choice;
}

// Main function to run the program
int main() {
    StudentDatabase db;  // Create database object
    int id;
    string name;
    float grade;

    while (true) {
        int choice = getMenuChoice();  // Get user choice
        cin.clear();                   // Clear any input errors
        cin.ignore(10000, '\n');       // Clear input buffer

        if (choice == 1) {  // Add student
            cout << "Enter ID: ";
            cin >> id;
            cin.ignore();  // Clear newline
            if (cin.fail()) {
                cout << "Invalid ID!" << endl;
                cin.clear();
                cin.ignore(10000, '\n');
                continue;
            }
            cout << "Enter Name: ";
            getline(cin, name);
            cout << "Enter Grade (0.0-100.0): ";
            cin >> grade;
            if (cin.fail()) {
                cout << "Invalid Grade!" << endl;
                cin.clear();
                cin.ignore(10000, '\n');
                continue;
            }
            db.addStudent(id, name, grade);  // Add to database
        }
        else if (choice == 2) {  // Display all
            db.displayAll();
        }
        else if (choice == 3) {  // Search by ID
            cout << "Enter ID to search: ";
            cin >> id;
            if (cin.fail()) {
                cout << "Invalid ID!" << endl;
                cin.clear();
                cin.ignore(10000, '\n');
                continue;
            }
            db.searchById(id);
        }
        else if (choice == 4) {  // Exit
            cout << "Exiting program." << endl;
            break;
        }
        else {
            cout << "Invalid choice! Try again." << endl;
        }
    }
    return 0;  // Successful execution
}
