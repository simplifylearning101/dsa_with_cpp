#include <iostream>
#include <fstream>
#include <vector>
#include <string>
#include <unordered_map>

// KMP Pattern Matching
std::vector<int> computeLPS(std::string pattern) {
    int m = pattern.length();
    std::vector<int> lps(m, 0);
    int len = 0, i = 1;
    while (i < m) {
        if (pattern[i] == pattern[len]) lps[i++] = ++len;
        else if (len != 0) len = lps[len - 1];
        else lps[i++] = 0;
    }
    return lps;
}
std::vector<int> KMPSearch(std::string text, std::string pattern) {
    int n = text.length(), m = pattern.length();
    std::vector<int> lps = computeLPS(pattern), result;
    int i = 0, j = 0;
    while (i < n) {
        if (text[i] == pattern[j]) { i++; j++; }
        if (j == m) { result.push_back(i - j); j = lps[j - 1]; }
        else if (i < n && text[i] != pattern[j]) {
            if (j != 0) j = lps[j - 1];
            else i++;
        }
    }
    return result;
}

// Group Anagrams
std::vector<std::vector<std::string>> groupAnagrams(std::vector<std::string>& strs) {
    std::unordered_map<std::string, std::vector<std::string>> map;
    for (std::string s : strs) {
        std::vector<int> count(26, 0);
        for (char c : s) count[c - 'a']++;
        std::string key;
        for (int x : count) key += std::to_string(x) + "#";
        map[key].push_back(s);
    }
    std::vector<std::vector<std::string>> result;
    for (auto& pair : map) result.push_back(pair.second);
    return result;
}

// Longest Palindromic Substring
std::string longestPalindrome(std::string s) {
    int n = s.length(), start = 0, maxLen = 1;
    for (int i = 0; i < n; i++) {
        int left = i, right = i;
        while (left >= 0 && right < n && s[left] == s[right]) {
            if (right - left + 1 > maxLen) {
                start = left;
                maxLen = right - left + 1;
            }
            left--; right++;
        }
        left = i; right = i + 1;
        while (left >= 0 && right < n && s[left] == s[right]) {
            if (right - left + 1 > maxLen) {
                start = left;
                maxLen = right - left + 1;
            }
            left--; right++;
        }
    }
    return s.substr(start, maxLen);
}

// String Compression
std::string compress(std::string chars) {
    int n = chars.length(), write = 0, count = 1, start = 0;
    std::string result = chars;
    for (int i = 1; i <= n; i++) {
        if (i < n && chars[i] == chars[i - 1]) {
            count++;
        } else {
            result[write++] = chars[start];
            if (count > 1) {
                std::string num = std::to_string(count);
                for (char c : num) result[write++] = c;
            }
            start = i;
            count = 1;
        }
    }
    return result.substr(0, write);
}

// Main Program
int main() {
    std::ifstream inFile("input.txt");
    std::ofstream outFile("output.txt");
    if (!inFile || !outFile) {
        std::cerr << "File I/O error" << std::endl;
        return 1;
    }

    // Read input
    std::string text, pattern;
    std::getline(inFile, text);
    std::getline(inFile, pattern);
    std::vector<std::string> words;
    std::string word;
    while (inFile >> word) words.push_back(word);

    // Task 1: Pattern Matching
    std::vector<int> matches = KMPSearch(text, pattern);
    outFile << "Pattern matches: [";
    for (int i = 0; i < matches.size(); i++) {
        outFile << matches[i];
        if (i < matches.size() - 1) outFile << ", ";
    }
    outFile << "]\n";

    // Task 2: Group Anagrams
    std::vector<std::vector<std::string>> anagrams = groupAnagrams(words);
    outFile << "Anagrams: [";
    for (int i = 0; i < anagrams.size(); i++) {
        outFile << "[";
        for (int j = 0; j < anagrams[i].size(); j++) {
            outFile << "\"" << anagrams[i][j] << "\"";
            if (j < anagrams[i].size() - 1) outFile << ", ";
        }
        outFile << "]";
        if (i < anagrams.size() - 1) outFile << ", ";
    }
    outFile << "]\n";

    // Task 3: Longest Palindromic Substring
    std::string palindrome = longestPalindrome(text);
    outFile << "Longest Palindrome: " << palindrome << "\n";

    // Task 4: Compress Text
    std::string compressed = compress(text);
    outFile << "Compressed: " << compressed << "\n";

    inFile.close();
    outFile.close();
    return 0;
}
