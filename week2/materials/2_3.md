# Solutions to Homework Problems

1. **Check if an array is a palindrome**
```cpp
#include <iostream>
bool isPalindrome(int arr[], int n) {
    int left = 0, right = n - 1;
    while (left < right) {
        if (arr[left] != arr[right]) return false;
        left++;
        right--;
    }
    return true;
}
int main() {
    int arr[] = {1, 2, 2, 1};
    std::cout << (isPalindrome(arr, 4) ? "Yes" : "No") << std::endl;
    return 0;
}
```

2. **Find two numbers in a sorted array that sum to a target**
```cpp
#include <iostream>
#include <algorithm>
std::pair<int, int> findPairSum(int arr[], int n, int target) {
    std::sort(arr, arr + n);
    int left = 0, right = n - 1;
    while (left < right) {
        int sum = arr[left] + arr[right];
        if (sum == target) return {left, right};
        else if (sum < target) left++;
        else right--;
    }
    return {-1, -1};
}
int main() {
    int arr[] = {2, 7, 11, 15};
    auto [i, j] = findPairSum(arr, 4, 9);
    std::cout << i << " " << j << std::endl; // Output: 0 1
    return 0;
}
```

3. **Remove duplicates from a sorted array in-place**
```cpp
#include <iostream>
int removeDuplicates(int arr[], int n) {
    if (n <= 1) return n;
    int i = 0;
    for (int j = 1; j < n; j++) {
        if (arr[j] != arr[i]) arr[++i] = arr[j];
    }
    return i + 1;
}
int main() {
    int arr[] = {1, 1, 2, 2, 3};
    int len = removeDuplicates(arr, 5);
    for (int i = 0; i < len; i++) {
        std::cout << arr[i] << " ";
    }
    std::cout << std::endl; // Output: 1 2 3
    return 0;
}
```

4. **Move all zeros to the end of an array in-place**
```cpp
#include <iostream>
void moveZeroes(int arr[], int n) {
    int nonZeroPos = 0;
    for (int i = 0; i < n; i++) {
        if (arr[i] != 0) arr[nonZeroPos++] = arr[i];
    }
    while (nonZeroPos < n) arr[nonZeroPos++] = 0;
}
int main() {
    int arr[] = {0, 1, 0, 3, 12};
    moveZeroes(arr, 5);
    for (int i = 0; i < 5; i++) {
        std::cout << arr[i] << " ";
    }
    std::cout << std::endl; // Output: 1 3 12 0 0
    return 0;
}
```

5. **Find the intersection of two sorted arrays**
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
std::vector<int> intersect(int arr1[], int n1, int arr2[], int n2) {
    std::sort(arr1, arr1 + n1);
    std::sort(arr2, arr2 + n2);
    std::vector<int> res;
    int i = 0, j = 0;
    while (i < n1 && j < n2) {
        if (arr1[i] == arr2[j]) {
            res.push_back(arr1[i]);
            i++;
            j++;
        } else if (arr1[i] < arr2[j]) i++;
        else j++;
    }
    return res;
}
int main() {
    int arr1[] = {1, 2, 2, 1}, arr2[] = {2, 2};
    auto res = intersect(arr1, 4, arr2, 2);
    for (int x : res) {
        std::cout << x << " "; // Output: 2 2
    }
    std::cout << std::endl;
    return 0;
}
```

6. **Reverse an array using two pointers**
```cpp
#include <iostream>
void reverseArray(int arr[], int n) {
    int left = 0, right = n - 1;
    while (left < right) {
        std::swap(arr[left++], arr[right--]);
    }
}
int main() {
    int arr[] = {1, 2, 3, 4, 5};
    reverseArray(arr, 5);
    for (int i = 0; i < 5; i++) {
        std::cout << arr[i] << " "; // Output: 5 4 3 2 1
    }
    std::cout << std::endl;
    return 0;
}
```

7. **Find the longest subarray with sum less than or equal to k**
```cpp
#include <iostream>
int longestSubarray(int arr[], int n, int k) {
    int sum = 0, maxLen = 0, left = 0;
    for (int right = 0; right < n; right++) {
        sum += arr[right];
        while (sum > k && left <= right) {
            sum -= arr[left++];
        }
        maxLen = std::max(maxLen, right - left + 1);
    }
    return maxLen;
}
int main() {
    int arr[] = {1, 2, 3, 4};
    std::cout << longestSubarray(arr, 4, 5) << std::endl; // Output: 2
    return 0;
}
```

8. **Partition array around a pivot**
```cpp
#include <iostream>
int partition(int arr[], int n, int pivot) {
    int i = 0;
    for (int j = 0; j < n; j++) {
        if (arr[j] <= pivot) std::swap(arr[i++], arr[j]);
    }
    return i;
}
int main() {
    int arr[] = {3, 1, 4, 1, 5};
    partition(arr, 5, 3);
    for (int i = 0; i < 5; i++) {
        std::cout << arr[i] << " "; // Output: 1 1 3 4 5
    }
    std::cout << std::endl;
    return 0;
}
```

9. **Merge two sorted arrays into a new sorted array**
```cpp
#include <iostream>
#include <vector>
std::vector<int> mergeSorted(int arr1[], int n1, int arr2[], int n2) {
    std::vector<int> res;
    int i = 0, j = 0;
    while (i < n1 && j < n2) {
        if (arr1[i] <= arr2[j]) res.push_back(arr1[i++]);
        else res.push_back(arr2[j++]);
    }
    while (i < n1) res.push_back(arr1[i++]);
    while (j < n2) res.push_back(arr2[j++]);
    return res;
}
int main() {
    int arr1[] = {1, 3, 5}, arr2[] = {2, 4, 6};
    auto res = mergeSorted(arr1, 3, arr2, 3);
    for (int x : res) {
        std::cout << x << " "; // Output: 1 2 3 4 5 6
    }
    std::cout << std::endl;
    return 0;
}
```

10. **Find if there exists a triplet summing to a target**
```cpp
#include <iostream>
#include <algorithm>
bool findTriplet(int arr[], int n, int target) {
    std::sort(arr, arr + n);
    for (int i = 0; i < n - 2; i++) {
        int left = i + 1, right = n - 1;
        while (left < right) {
            int sum = arr[i] + arr[left] + arr[right];
            if (sum == target) return true;
            else if (sum < target) left++;
            else right--;
        }
    }
    return false;
}
int main() {
    int arr[] = {1, 4, 6, 8, 10};
    std::cout << (findTriplet(arr, 5, 12) ? "Yes" : "No") << std::endl; // Output: Yes
    return 0;
}
```
